常见使用场景和完整代码示例

🔰 基础使用

🔍

搜索和浏览内容

快速查找和过滤您的内容库

// 场景1: 搜索所有包含 "AI" 的内容
content_system_search({
  "query": "AI"
})

// 场景2: 仅搜索作品类型
content_system_search({
  "type": "work",
  "limit": 20
})

// 场景3: 按年份和标签组合搜索
content_system_search({
  "type": "node",
  "year": 2024,
  "tags": ["展览", "新媒体艺术"]
})

// 场景4: 分页浏览大量内容
content_system_search({
  "type": "asset",
  "limit": 50,
  "offset": 100
})

创建各种类型的内容节点

    "status": "published"
  }

    "tags": ["艺术家", "数字媒体", "交互"],

  }
})

// 创建一个展览节点
content_system_create({
  "type": "node",

  }
})

🔗 建立关联关系

👥

作者与作品的关联

将艺术家与他们的作品关联起来

// 首先创建人物和作品
const personResult = content_system_create({
  "type": "person",
  "fields": {
    "name": "Ewan Qian",
    "summary": "多媒体艺术家"
  }
})

const workResult = content_system_create({
  "type": "work",
  "fields": {
    "title": "Drop Flow",
    "description": "交互式数字艺术作品"
  }
})

// 建立作者关系
content_system_link({
  "source_id": personResult.id,
  "target_id": workResult.id,
  "relation_type": "author_of",
  "fields": {
    "description": "主要创作者",
    "start_date": "2023"
  }
})

// 查看完整信息
content_system_get({
  "id": workResult.id
})
🖼️

作品在展览中展出

建立作品与展览节点的关联

// 创建展览节点
const nodeResult = content_system_create({
  "type": "node",
  "fields": {
    "title": "杭州开幕展",
    "date": "2024-03"
  }
})

// 将作品关联到展览
content_system_link({
  "source_id": "work:drop-flow",
  "target_id": nodeResult.id,
  "relation_type": "featured_in",
  "fields": {
    "description": "首次展出",
    "notes": "主展厅中央位置"
  }
})

// 也可以建立双向关联查看
content_system_link({
  "source_id": nodeResult.id,
  "target_id": "work:drop-flow",
  "relation_type": "features",
  "fields": {
    "description": "精选参展作品"
  }
})

🚀 完整工作流

🏗️

创建作品并建立完整关联

从创建到关联的完整流程

// ==========================================
// 场景: 创建一个新作品并关联到人物和展览
// ==========================================

// 1. 创建艺术家
const artistResult = content_system_create({
  "type": "person",
  "fields": {
    "name": "Ewan Qian",
    "summary": "多媒体艺术家，专注于数字艺术与交互设计",
    "tags": ["艺术家", "数字媒体", "交互"]
  }
})

// 2. 创建协作团队
const collectiveResult = content_system_create({
  "type": "collective",
  "fields": {
    "name": "VIRTURA 艺术实验室",
    "summary": "跨学科艺术与技术研究团队",
    "tags": ["团队", "研究", "艺术+科技"]
  }
})

// 3. 创建新作品
const workResult = content_system_create({
  "type": "work",
  "fields": {
    "title": "Digital Dreams",
    "description": "An interactive digital installation exploring AI creativity",
    "summary": "通过算法生成与人工干预相结合的方式，探索新的艺术表达方式。",
    "tags": ["数字艺术", "AI生成", "交互装置"],
    "date": "2024",
    "status": "published"
  }
})

// 4. 创建展览节点
const exhibitionResult = content_system_create({
  "type": "node",
  "fields": {
    "title": "杭州开幕展",
    "description": "新媒体艺术与技术融合展览",
    "tags": ["展览", "杭州", "新媒体艺术"],
    "date": "2024-03",
    "status": "completed"
  }
})

// 5. 关联作品到作者
content_system_link({
  "source_id": artistResult.id,
  "target_id": workResult.id,
  "relation_type": "author_of",
  "fields": {
    "description": "主要创作者",
    "start_date": "2023"
  }
})

// 6. 关联作品到团队
content_system_link({
  "source_id": collectiveResult.id,
  "target_id": workResult.id,
  "relation_type": "contributor_to",
  "fields": {
    "description": "技术支持与协作"
  }
})

// 7. 关联作品到展览
content_system_link({
  "source_id": workResult.id,
  "target_id": exhibitionResult.id,
  "relation_type": "featured_in",
  "fields": {
    "description": "首次展出",
    "location": "主展厅"
  }
})

// 8. 获取作品完整信息，包含所有关联
const completeWork = content_system_get({
  "id": workResult.id,
  "include_relations": true
})

console.log("完整作品信息:", completeWork)

批量导出内容

将内容库导出为不同格式

content_system_export({
  "type": "work",
  "format": "json"
})

// 导出特定对象为Markdown
content_system_export({
  "ids": ["work:drop-flow", "work:timer", "node:hangzhou-opening"],
  "format": "markdown"
})

// 导出2024年的展览为CSV
content_system_export({
  "type": "node",
  "tags": ["展览"],
  "year": 2024,
  "format": "csv"
})

// 导出所有公开内容
content_system_export({
  "type": "asset",
  "tags": ["public"],
  "format": "json",
  "limit": 100
})

💻 CLI 命令行使用

⚡

常用CLI命令

通过命令行管理内容库

# 列出所有作品
./weaver list --type work

# 列出特定类型，带标签过滤
./weaver list --type node --tags "展览,2024"

# 创建新节点
./weaver create --type node \
  --title "AI生成工作流" \
  --tags "AI,工作流" \
  --description "使用AI进行内容创作的工作流程"

# 查看对象详情
./weaver show --id work:drop-flow

# 建立关联
./weaver link \
  --object1 work:drop-flow \
  --object2 node:ai-generation-workflow \
  --relation uses_method

# 导出内容
./weaver export --type work --format json

# 软删除
./weaver delete --id work:test-work

投料工厂流程

批量导入和处理内容

# 把素材放入incoming目录
mkdir -p incoming/
cp ~/Downloads/images/* incoming/

# 批量导入，生成待审核清单
python3 scripts/factory_ingest.py import --dir incoming/

# 查看待审核项目
python3 scripts/factory_ingest.py review --list

# 审核确认
python3 scripts/factory_ingest.py review --confirm "批准所有"

# 也可以使用自然语言指令
python3 scripts/factory_ingest.py review --confirm "批准2024年的图片，其他待复核"

# 一键运行全流水线
python3 scripts/build_pipeline.py

# 启动本地服务查看
python3 -m http.server 8080
# 打开浏览器: http://localhost:8080/viewer/

构建知识图谱

创建丰富的内容关联网络

// 构建一个艺术项目的完整知识图谱

// 1. 创建核心概念
content_system_create({
  "type": "node",
  "fields": {
    "title": "AI艺术概念",
    "summary": "探索人工智能与艺术融合的核心概念",
    "tags": ["概念", "理论", "AI艺术"]
  }
})

// 2. 创建方法流程
content_system_create({
  "type": "node",
  "fields": {
    "title": "生成式工作流",
    "summary": "使用生成模型创作艺术的标准流程",
    "tags": ["方法", "工作流", "生成式AI"]
  }
})

// 3. 创建工具和规格
content_system_create({
  "type": "spec",
  "fields": {
    "title": "生成模型规格",
    "summary": "推荐使用的AI模型和技术规格",
    "tags": ["规格", "技术", "AI模型"]
  }
})

// 4. 建立概念之间的关联
content_system_link({
  "source_id": "node:ai-art-concept",
  "target_id": "node:generative-workflow",
  "relation_type": "references",
  "fields": {
    "description": "概念通过工作流实现"
  }
})

content_system_link({
  "source_id": "node:generative-workflow",
  "target_id": "spec:generative-model-spec",
  "relation_type": "uses",
  "fields": {
    "description": "工作流使用此规格的模型"
  }
})

// 5. 将具体作品关联到概念
content_system_link({
  "source_id": "work:digital-dreams",
  "target_id": "node:ai-art-concept",
  "relation_type": "part_of",
  "fields": {
    "description": "作品是AI艺术概念的实践"
  }
})

content_system_link({
  "source_id": "work:digital-dreams",
  "target_id": "node:generative-workflow",
  "relation_type": "uses_method",
  "fields": {
    "description": "使用生成式工作流创作"
  }
})
